function placeOrder() {
let userName =
document.getElementById('name').value;
let qty = document.getElementById('quantity').value;
let product = document.getElementById('product').value;
if(userName===""|| product===""|| qty==="") {
alert("Please fill in all fields.");
return;
}
//save order

localStorage.setItem('order', JSON.stringify({userName, product, qty}));
//new Order object
let newOrder = {
name: userName,
product: product,
quantity: qty
Date: new Date().toISOString()
};
//old orders from localStorage
let oldOrders = JSON.parse(localStorage.getItem('orders')) || [];
//new order add
oldOrders.push(newOrder);
//save back to localStorage
localStorage.setItem('orders', JSON.stringify(oldOrders));
alert("Order placed successfully!");
window.location.href = "order-summary.html";
}
